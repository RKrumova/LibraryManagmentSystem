CREATE TABLE userinformation (
FNAME NVARCHAR(20) NOT NULL,
LNAME NVARCHAR(20) NOT NULL,
AGE int NOT NULL,

QUESTION VARCHAR(50) NOT NULL,
ANSWER VARCHAR(50) NOT NULL
);
